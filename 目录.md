<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>mapilot-gl-doc</title>
</head>

<body>
    <div id="back-content" style="position: fixed; left: 10px;top: 5px; font-size: 24px; cursor: pointer;"><a>返回</a>
    </div>
    <div id="markdown-content"></div>
    <link href="./css/bootstrap-3.3.5.css" rel="stylesheet" />
    <script src="./js/marked.min.js"></script>
    <link rel="stylesheet" href="./css/github.min.css">
    <script src="./js/highlight.min.js"></script>
    <style>
        /* Optional: Add some basic styling */
        body {
            font-family: Arial, sans-serif;
            margin: 20px;
            padding: 20px;
            text-align: center;
        }

        #markdown-content {
            max-width: 1000px;
            margin: 0 auto;
            text-align: left;
        }

        table {
            border-collapse: collapse;
            width: 100%;
        }

        th,
        td {
            border: 1px solid #ddd;
            padding: 8px;
            text-align: left;
        }

        th {
            background-color: #f2f2f2;
        }
    </style>
    <script>

        document.getElementById('back-content').addEventListener('click', (event) => {
            event.preventDefault(); // 阻止默认行为
            location.reload();
        });



        // 加载 Markdown 文件
        fetch('../目录.md')
            .then(response => response.text())
            .then(markdown => {
                // 启用表格渲染
                const rendererMD = new marked.Renderer()
                marked.setOptions({
                    renderer: rendererMD,
                    gfm: true,
                    tables: true,
                    breaks: false,
                    pedantic: false,
                    sanitize: false,
                    smartLists: true,
                    smartypants: false,
                    highlight: function (code, lang) {
                        // 使用 highlight.js 来对代码进行高亮
                        return lang ? hljs.highlight(lang, code).value : hljs.highlightAuto(code).value;
                    },
                })

                document.getElementById('markdown-content').innerHTML = marked(markdown);

                // 处理超链接，点击后动态加载目标文件并显示
                const links = Array.from(document.querySelectorAll('a'));
                links.forEach(link => {
                    link.addEventListener('click', (event) => {
                        event.preventDefault(); // 阻止默认行为
                        const href = event.target.getAttribute('href');
                        fetch(href)
                            .then(response => response.text())
                            .then(markdown => {
                                document.getElementById('markdown-content').innerHTML = marked(markdown);
                            })
                            .catch(error => console.error('Error loading file:', error));
                    });
                });
            })
            .catch(error => console.error('Error loading file:', error));
    </script>
</body>

</html>
